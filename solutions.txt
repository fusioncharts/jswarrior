===============

jsWarrior.turn = function(warrior) {
  warrior.last_known_health = warrior.last_known_health ? warrior.last_known_health : warrior.health;
  if(warrior.check() == "enemy")
  {
    warrior.attack();
  }
  else{
    if(warrior.health < warrior.last_known_health)
    {
      warrior.walk();
    }
    else if(warrior.health < 15)
    {
      warrior.rest();
    }
    else {
	  warrior.walk();
    }
  }
  warrior.last_known_health = warrior.health;
}

Level 6:
==============

jsWarrior.turn = function(warrior) {
  warrior.last_known_health = warrior.last_known_health ? warrior.last_known_health : warrior.health;
  if(warrior.check() == "enemy")
  {
    warrior.attack();
  }
  else if(warrior.check() == "captive")
  {
    warrior.rescue();
  }
  else{
    if(warrior.health < warrior.last_known_health)
    {
      if(warrior.health < 8)
	    warrior.walk("backward");
      else
        warrior.walk("forward");
    }
    else
    {
      if(warrior.health < 15)
      {
        warrior.rest();
      }
      else{
        warrior.walk();
      }
    }
  }
  warrior.last_known_health = warrior.health;
}

jsWarrior.turn = function(warrior) {
  warrior.last_known_health = warrior.last_known_health ? warrior.last_known_health : warrior.health;

  if(warrior.check() == "wall")
  {
    warrior.pivot();
  }
  else if(warrior.check() == "enemy")
  {
    warrior.attack();
  }
  else if(warrior.check() == "captive")
  {
    warrior.rescue();
  }
  else{
    if(warrior.health < warrior.last_known_health)
    {
      if(warrior.health < 8)
	    warrior.walk("backward");
      else
        warrior.walk("forward");
    }
    else
    {
      if(warrior.health < 15)
      {
        warrior.rest();
      }
      else{
        warrior.walk();
      }
    }
  }
  warrior.last_known_health = warrior.health;
}
